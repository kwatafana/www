     40 impl DB for SqliteDB {

     48     fn connect(&mut self) -> Result<(), DatabaseError> {
     49         // Open database connection
     50         let conn = Connection::open(self.path.clone())?;
     51         self.conn = Some(conn);
     52 
     53         // Create accounts table if it does not already exists
     54         match &self.conn {
     55             Some(conn) => match conn.execute(accounts_sql::CREATE_TABLE, []) {
     56                 Ok(_rows) => Ok(()),
     57                 Err(_err) => Err(DatabaseError::CreateTableError),
     58             },
     59             None => Err(DatabaseError::ConnectionError),
     60         }
     61     }
     62 
     63     fn store_account(&self, account: Account) -> Result<(), DatabaseError> {
     64         match &self.conn {
     65             Some(conn) => {
     66                 conn.execute(
     67                     accounts_sql::STORE,
     68                     params![account.username, account.password, account.email],
     69                 )?;
     70                 Ok(())
     71             }
     72             None => Err(DatabaseError::ConnectionError),
     73         }
     74     }
     75 
     76     fn get_public_account(&self, username: &str) -> Result<PublicAccount, DatabaseError> {
     77         match &self.conn {
     78             Some(conn) => {
     79                 let mut stmt = conn.prepare(accounts_sql::GET_PUBLIC)?;
     80                 let mut rows = stmt.query(&[(":username", username)])?;
     81                 match rows.next()? {
     82                     Some(s) => Ok(PublicAccount {
     83                         username: s.get(0)?,
     84                     }),
     85                     None => Err(DatabaseError::AccountRetrievalError),
     86                 }
     87             }
     88             None => Err(DatabaseError::ConnectionError),
     89         }
     90     }
     91 }
