     33         let account = account::Account::create(payload)?;
     34         // TODO: email verification

     36         // store account in database
     37         if cfg!(sqlite) {
     38             if let Some(db) = &self.database {
     39                 db.account_store(&account)?;
     40             }
     41         }

     45     /// Get account public data, using the username as ID

     47         if cfg!(sqlite) {
     48             if let Some(db) = &self.database {
     49                 let public_account = db.account_get_pub(username)?;
     50                 Ok(public_account)
     51             } else {
     52                 Err(database::error::Error::Connection)
     53             }
     54         } else {
     55             Err(database::error::Error::Connection)
